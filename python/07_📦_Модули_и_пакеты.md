# 07_📦_Модули_и_пакеты

## Философия модулей в Python
- **Один файл = один модуль**
- **Папка с `__init__.py` = пакет**
- **Всё — объект** (модули тоже объекты)
- **Пространства имен** — модули создают отдельное пространство имен
- **"Батарейки включены"** — огромная стандартная библиотека

---

## Импорт модулей

### Основные способы импорта

```python
# 1. Импорт всего модуля
import math
print(math.pi)
print(math.sqrt(16))

# 2. Импорт конкретных объектов
from math import pi, sqrt
print(pi)
print(sqrt(16))

# 3. Импорт с псевдонимом
import math as m
print(m.pi)

from math import sqrt as square_root
print(square_root(16))

# 4. Импорт всего (ЗЛО! НЕ ДЕЛАЙ ТАК!)
from math import *  # загрязняет пространство имен
# Теперь неясно, откуда взялись pi, sin, cos...
```

### Что можно импортировать?
```python
# Функции
from math import factorial

# Классы
from collections import defaultdict, Counter

# Переменные/константы
from math import pi, e

# Модули целиком
import os
import sys
import json

# Подмодули
import os.path
from os.path import join, exists
```

## Создание своего модуля

### Простой модуль (файл mymath.py)
```python
# Файл: mymath.py
"""Мой математический модуль"""

PI = 3.14159
E = 2.71828

def add(a, b):
    """Сложение двух чисел"""
    return a + b

def subtract(a, b):
    """Вычитание"""
    return a - b

def circle_area(radius):
    """Площадь круга"""
    return PI * radius ** 2

class Calculator:
    """Простой калькулятор"""
    def __init__(self, initial=0):
        self.value = initial
    
    def add(self, x):
        self.value += x
        return self.value
    
    def reset(self):
        self.value = 0
```

### Использование модуля (файл main.py)
```python
# main.py
import mymath

print(mymath.PI)                    # 3.14159
print(mymath.add(5, 3))              # 8
print(mymath.circle_area(10))        # 314.159

calc = mymath.Calculator(10)
print(calc.add(5))                   # 15

# Или так
from mymath import add, Calculator
print(add(5, 3))
calc = Calculator(10)
```

## Специальная переменная `__name__`

### Зачем это нужно?
```python
# Файл: mymodule.py
def main():
    print("Запуск основной функции")
    # здесь основной код

if __name__ == "__main__":
    # Этот код выполнится ТОЛЬКО если файл запущен напрямую
    # НЕ выполнится при импорте
    print("Файл запущен напрямую")
    main()
else:
    print(f"Модуль {__name__} импортирован")
```

### Практический пример
```python
# Файл: utils.py
def parse_data(data):
    # полезная функция
    return processed_data

def test_parse():
    # тесты для функции
    assert parse_data("test") == expected

if __name__ == "__main__":
    # Запуск тестов при прямом вызове
    test_parse()
    print("Все тесты пройдены!")
```

## Создание пакета

### Структура пакета
```
mypackage/                  # папка пакета
├── __init__.py             # обязательный файл
├── module1.py              # модуль 1
├── module2.py              # модуль 2
└── subpackage/             # подпакет
    ├── __init__.py
    └── module3.py
```

### Файл `__init__.py`
```python
# mypackage/__init__.py

# 1. Может быть пустым (просто маркер пакета)

# 2. Может инициализировать пакет
print("Инициализация пакета mypackage")

# 3. Может определять, что импортируется при from package import *
__all__ = ['module1', 'module2']  # только эти модули

# 4. Может импортировать удобные функции наверх
from .module1 import useful_function
from .module2 import AnotherClass

# Теперь можно:
# from mypackage import useful_function
```

### Модули пакета
```python
# mypackage/module1.py
def useful_function():
    return "Полезная функция из module1"

# mypackage/module2.py
class AnotherClass:
    def __init__(self, name):
        self.name = name
    
    def greet(self):
        return f"Привет от {self.name}"

# mypackage/subpackage/module3.py
def deep_function():
    return "Глубоко в подпакете"
```

### Использование пакета
```python
# Разные способы импорта
import mypackage.module1
from mypackage import module2
from mypackage.subpackage import module3
from mypackage.module1 import useful_function
from mypackage.module2 import AnotherClass

# Использование
mypackage.module1.useful_function()
obj = AnotherClass("тест")
module3.deep_function()
```

## Абсолютный и относительный импорт

### Абсолютный импорт (рекомендуется)
```python
# Внутри пакета
from mypackage.module1 import useful_function
from mypackage.subpackage.module3 import deep_function
```

### Относительный импорт (только внутри пакета)
```python
# Внутри mypackage/module2.py

# Импорт из того же пакета
from .module1 import useful_function      # module1 в той же папке
from .subpackage import module3           # подпакет

# Импорт из родительского пакета
from ..module1 import useful_function     # на уровень выше
from ..subpackage.module3 import deep_function

# Импорт из "сестры"
from ..otherpackage import something
```

**Правила:**
- `.` — текущий пакет
- `..` — родительский пакет
- `...` — прародительский (редко)
- Нельзя использовать в скриптах верхнего уровня

## Поиск модулей (sys.path)

### Где Python ищет модули?
```python
import sys

# Список путей поиска
print(sys.path)

# Типичные пути:
# 1. Директория текущего скрипта
# 2. PYTHONPATH (переменная окружения)
# 3. Стандартные пути установки Python
# 4. Пути из site-packages (сторонние библиотеки)
```

### Добавление своих путей
```python
import sys

# Добавить путь (временно, для текущей сессии)
sys.path.append("/path/to/my/modules")
sys.path.insert(0, "/another/path")  # в начало списка

# Теперь можно импортировать оттуда
import my_custom_module

# Лучше использовать PYTHONPATH или установку пакета
```

### Переменная окружения PYTHONPATH
```bash
# Linux/Mac
export PYTHONPATH="/path/to/modules:$PYTHONPATH"
python my_script.py

# Windows (cmd)
set PYTHONPATH=C:\path\to\modules;%PYTHONPATH%
python my_script.py
```

## Сторонние модули и pip

### Установка пакетов
```bash
# Установка одного пакета
pip install requests

# Установка конкретной версии
pip install requests==2.28.1

# Установка нескольких пакетов
pip install numpy pandas matplotlib

# Установка из requirements.txt
pip install -r requirements.txt

# Обновление пакета
pip install --upgrade requests
```

### Управление пакетами
```bash
# Список установленных пакетов
pip list

# Информация о пакете
pip show requests

# Поиск пакета
pip search "web framework"

# Удаление пакета
pip uninstall requests

# Заморозка зависимостей (для проекта)
pip freeze > requirements.txt
# requirements.txt:
# requests==2.28.1
# numpy==1.24.2
# pandas==1.5.3
```

### Виртуальное окружение (обязательно для проектов!)
```bash
# Создание
python -m venv venv

# Активация (Linux/Mac)
source venv/bin/activate

# Активация (Windows)
venv\Scripts\activate

# Деактивация
deactivate

# Установка пакетов в виртуальное окружение
pip install requests

# Сохранение зависимостей
pip freeze > requirements.txt
```

## Популярные встроенные модули

### sys — системные параметры
```python
import sys

# Аргументы командной строки
print(sys.argv)  # ['script.py', 'arg1', 'arg2']

# Выход из программы
sys.exit(0)      # 0 - успех, 1 - ошибка

# Информация о Python
print(sys.version)
print(sys.platform)  # 'win32', 'linux', 'darwin'

# Пути поиска модулей
print(sys.path)

# Стандартный ввод/вывод
sys.stdout.write("Hello\n")
data = sys.stdin.readline()
```

### os — операционная система
```python
import os

# Текущая директория
os.getcwd()
os.chdir("/path")

# Переменные окружения
os.environ.get('HOME')
os.environ['MY_VAR'] = 'value'

# Запуск команд
os.system('ls -la')
os.system('dir')  # на Windows

# Информация о системе
os.name          # 'posix', 'nt'
os.cpu_count()   # количество ядер
os.getpid()      # ID процесса
```

### shutil — операции с файлами
```python
import shutil

# Копирование
shutil.copy('src.txt', 'dst.txt')
shutil.copytree('src_dir', 'dst_dir')

# Перемещение
shutil.move('file.txt', 'archive/file.txt')

# Удаление
shutil.rmtree('dir_with_stuff')

# Архивация
shutil.make_archive('backup', 'zip', 'myfolder')
shutil.unpack_archive('backup.zip')
```

### glob — поиск файлов по маске
```python
import glob

# Все .txt файлы в текущей папке
txt_files = glob.glob("*.txt")

# Рекурсивно все .py файлы
py_files = glob.glob("**/*.py", recursive=True)

# Смешанные паттерны
images = glob.glob("*.{jpg,jpeg,png,gif}")
```

## Создание устанавливаемого пакета

### Структура проекта
```
mypackage/
├── setup.py                 # файл установки
├── README.md                # описание
├── LICENSE                  # лицензия
├── requirements.txt         # зависимости
├── mypackage/               # сам пакет
│   ├── __init__.py
│   ├── module1.py
│   └── module2.py
└── tests/                   # тесты
    ├── __init__.py
    └── test_module1.py
```

### setup.py
```python
from setuptools import setup, find_packages

setup(
    name='mypackage',
    version='0.1.0',
    author='Ваше Имя',
    author_email='your@email.com',
    description='Краткое описание пакета',
    long_description=open('README.md').read(),
    long_description_content_type='text/markdown',
    url='https://github.com/username/mypackage',
    packages=find_packages(),
    classifiers=[
        'Programming Language :: Python :: 3',
        'License :: OSI Approved :: MIT License',
        'Operating System :: OS Independent',
    ],
    python_requires='>=3.6',
    install_requires=[
        'requests>=2.25.0',
        'numpy>=1.19.0',
    ],
    extras_require={
        'dev': ['pytest', 'flake8'],
        'plot': ['matplotlib'],
    },
    entry_points={
        'console_scripts': [
            'mycommand=mypackage.cli:main',
        ],
    },
)
```

### Установка пакета
```bash
# Установка в режиме разработки
pip install -e .

# Сборка дистрибутива
python setup.py sdist bdist_wheel

# Установка из собранного дистрибутива
pip install dist/mypackage-0.1.0-py3-none-any.whl
```

## Полезные трюки

### Перезагрузка модуля
```python
import importlib
import mymodule

# После изменений в mymodule
importlib.reload(mymodule)
```

### Проверка, откуда импортирован модуль
```python
import math
print(math.__file__)      # путь к файлу модуля
print(math.__name__)       # имя модуля
print(math.__package__)    # имя пакета
```

### Ленивый импорт (для ускорения)
```python
def use_math():
    # Импорт внутри функции - только при вызове
    import math
    return math.pi

def use_heavy_module():
    from heavy_module import HeavyClass
    return HeavyClass()
```

```python
import sys

if sys.platform == 'win32':
    import msvcrt
    def get_key():
        return msvcrt.getch()
else:
    import tty
    import termios
    def get_key():
        # Unix способ
        pass
```

## Сравнение с C++

| Концепция | C++ | Python |
|-----------|-----|--------|
| Подключение | `#include <header>` | `import module` |
| Свои заголовки | `#include "myheader.h"` | `import mymodule` |
| Пространства имен | `namespace name { ... }` | Модули как пространства имен |
| Разделение | .h и .cpp файлы | Один .py файл |
| Библиотеки | Статические/динамические | Пакеты PyPI |
| Инструмент | vcpkg/conan | pip |
| Сборка | Компиляция | Интерпретация |
