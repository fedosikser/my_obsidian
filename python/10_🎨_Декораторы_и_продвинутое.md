# 10_🎨_Декораторы_и_продвинутое

## Философия декораторов
- **Функции — объекты первого класса** (их можно передавать и возвращать)
- **Декоратор** — функция, которая принимает функцию и возвращает функцию
- **Синтаксический сахар** — `@decorator` над определением функции
- **DRY** (Don't Repeat Yourself) — выносим повторяющийся код

## Основы декораторов

### Функции как объекты
```python
def say_hello(name):
    return f"Привет, {name}!"

# Функцию можно присвоить переменной
greet = say_hello
print(greet("Анна"))  # Привет, Анна!

# Функцию можно передать в другую функцию
def call_twice(func, arg):
    return func(arg), func(arg)

result1, result2 = call_twice(say_hello, "Боб")
print(result1, result2)  # Привет, Боб! Привет, Боб!

# Функцию можно вернуть из функции
def create_multiplier(n):
    def multiplier(x):
        return x * n
    return multiplier

double = create_multiplier(2)
triple = create_multiplier(3)
print(double(5))  # 10
print(triple(5))  # 15
```

### Простейший декоратор
```python
def simple_decorator(func):
    print("Декоратор работает!")
    return func  # возвращаем функцию без изменений

@simple_decorator
def hello():
    print("Привет!")

# При определении функции сразу сработает print
# Декоратор работает!
```

### Декоратор, изменяющий поведение
```python
def announce(func):
    def wrapper():
        print("Функция запускается...")
        result = func()
        print("Функция завершилась")
        return result
    return wrapper

@announce
def hello():
    print("Привет, мир!")
    return 42

# Вызов
result = hello()
# Функция запускается...
# Привет, мир!
# Функция завершилась
print(result)  # 42

# Что на самом деле происходит:
# hello = announce(hello)
```

### Декоратор с аргументами функции
```python
def logger(func):
    def wrapper(*args, **kwargs):
        print(f"Вызов {func.__name__} с args={args}, kwargs={kwargs}")
        result = func(*args, **kwargs)
        print(f"Результат: {result}")
        return result
    return wrapper

@logger
def add(a, b):
    return a + b

@logger
def greet(name, greeting="Привет"):
    return f"{greeting}, {name}!"

add(3, 5)
# Вызов add с args=(3,5), kwargs={}
# Результат: 8

greet("Анна", greeting="Здравствуй")
# Вызов greet с args=('Анна',), kwargs={'greeting': 'Здравствуй'}
# Результат: Здравствуй, Анна!
```

## Сохранение метаданных (functools.wraps)

### Проблема: потеря имени и документации
```python
def decorator(func):
    def wrapper(*args, **kwargs):
        """Внутренняя функция-обертка"""
        return func(*args, **kwargs)
    return wrapper

@decorator
def hello():
    """Документация функции hello"""
    print("Привет!")

print(hello.__name__)  # 'wrapper' (не 'hello'!)
print(hello.__doc__)   # 'Внутренняя функция-обертка' (не док функции!)
```

### Решение: @wraps
```python
from functools import wraps

def decorator(func):
    @wraps(func)  # копирует имя, docstring и другие атрибуты
    def wrapper(*args, **kwargs):
        """Внутренняя функция-обертка"""
        return func(*args, **kwargs)
    return wrapper

@decorator
def hello():
    """Документация функции hello"""
    print("Привет!")

print(hello.__name__)  # 'hello'
print(hello.__doc__)   # 'Документация функции hello'
```

## Полезные декораторы

### Таймер (измерение времени)
```python
import time
from functools import wraps

def timer(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        end = time.perf_counter()
        print(f"{func.__name__} выполнялась {end - start:.4f} сек")
        return result
    return wrapper

@timer
def slow_function():
    time.sleep(1)
    return "Готово"

slow_function()  # slow_function выполнялась 1.0002 сек
```

### Логирование вызовов
```python
def log_calls(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        args_repr = [repr(a) for a in args]
        kwargs_repr = [f"{k}={v!r}" for k, v in kwargs.items()]
        signature = ", ".join(args_repr + kwargs_repr)
        print(f"→ {func.__name__}({signature})")
        
        result = func(*args, **kwargs)
        
        print(f"← {func.__name__} вернула {result!r}")
        return result
    return wrapper

@log_calls
def multiply(x, y):
    return x * y

multiply(3, 5)
# → multiply(3, 5)
# ← multiply вернула 15
```

### Retry (повтор при ошибке)
```python
def retry(max_attempts=3, delay=1):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_attempts:
                        print(f"Все {max_attempts} попыток провалились")
                        raise
                    print(f"Попытка {attempt} не удалась: {e}. Пробуем снова...")
                    time.sleep(delay)
            return None
        return wrapper
    return decorator

@retry(max_attempts=3, delay=0.5)
def unstable_network_call():
    import random
    if random.random() < 0.7:
        raise ConnectionError("Network error")
    return "Success"

unstable_network_call()
```

### Кэширование (мемоизация)
```python
def memoize(func):
    cache = {}
    
    @wraps(func)
    def wrapper(*args):
        if args in cache:
            print(f"Берем из кэша: {args}")
            return cache[args]
        print(f"Вычисляем: {args}")
        result = func(*args)
        cache[args] = result
        return result
    return wrapper

@memoize
def fibonacci(n):
    if n < 2:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

fibonacci(10)  # вычисляется
fibonacci(10)  # берется из кэша
```

### Проверка типов (простая)
```python
def type_check(*arg_types, return_type=None):
    def decorator(func):
        @wraps(func)
        def wrapper(*args):
            # Проверка аргументов
            for i, (arg, expected) in enumerate(zip(args, arg_types)):
                if not isinstance(arg, expected):
                    raise TypeError(
                        f"Аргумент {i} должен быть {expected.__name__}, "
                        f"получен {type(arg).__name__}"
                    )
            
            result = func(*args)
            
            # Проверка возвращаемого значения
            if return_type and not isinstance(result, return_type):
                raise TypeError(
                    f"Возвращаемое значение должно быть {return_type.__name__}, "
                    f"получен {type(result).__name__}"
                )
            return result
        return wrapper
    return decorator

@type_check(int, int, return_type=int)
def add(a, b):
    return a + b

add(3, 5)    # OK
# add(3, "5")  # TypeError!
```

### Декоратор для отладки
```python
def debug(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        print(f"--- Отладка {func.__name__} ---")
        print(f"Позиционные аргументы: {args}")
        print(f"Именованные аргументы: {kwargs}")
        
        result = func(*args, **kwargs)
        
        print(f"Результат: {result}")
        print(f"--- Конец отладки {func.__name__} ---")
        return result
    return wrapper

@debug
def process_data(name, age, **kwargs):
    return f"{name} ({age})"

process_data("Анна", 30, city="Москва", job="Инженер")
```

## Декораторы с аргументами

### Декоратор, принимающий параметры
```python
def repeat(n_times):
    """Декоратор, повторяющий вызов функции n раз"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            results = []
            for _ in range(n_times):
                results.append(func(*args, **kwargs))
            return results
        return wrapper
    return decorator

@repeat(3)
def say(word):
    return word

print(say("Привет"))  # ['Привет', 'Привет', 'Привет']
```

### Декоратор с необязательными аргументами
```python
def repeat(arg=None):
    """Может быть вызван как @repeat или @repeat(3)"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            n = arg if callable(arg) else (arg or 1)
            results = []
            for _ in range(n if isinstance(n, int) else 1):
                results.append(func(*args, **kwargs))
            return results if len(results) > 1 else results[0]
        return wrapper
    
    # Если arg - функция, вызываем сразу
    if callable(arg):
        return decorator(arg)
    return decorator

@repeat
def hello1():
    return "Привет"

@repeat(3)
def hello2():
    return "Привет"

print(hello1())  # 'Привет'
print(hello2())  # ['Привет', 'Привет', 'Привет']
```

```python
def bold(func):
    @wraps(func)
    def wrapper():
        return f"<b>{func()}</b>"
    return wrapper

def italic(func):
    @wraps(func)
    def wrapper():
        return f"<i>{func()}</i>"
    return wrapper

@bold
@italic

# Что происходит:
# 1. italic(hello) -> функция в <i>
# 2. bold(italic(hello)) -> оборачивает в <b>
```

## Декораторы классов

### Декоратор для добавления методов
```python
def add_repr(cls):
    """Добавляет методу __repr__ в класс"""
    def __repr__(self):
        attrs = ', '.join(f"{k}={v!r}" for k, v in self.__dict__.items())
        return f"{cls.__name__}({attrs})"
    
    cls.__repr__ = __repr__
    return cls

@add_repr
class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

p = Person("Анна", 30)
print(p)  # Person(name='Анна', age=30)
```

### Декоратор для синглтона
```python
def singleton(cls):
    """Превращает класс в синглтон"""
    instances = {}
    
    @wraps(cls)
    def get_instance(*args, **kwargs):
        if cls not in instances:
            instances[cls] = cls(*args, **kwargs)
        return instances[cls]
    
    return get_instance

@singleton
class Database:
    def __init__(self):
        print("Создание подключения к БД")
        self.connected = True

db1 = Database()  # Создание подключения к БД
db2 = Database()  # (ничего не печатает)
print(db1 is db2)  # True
```

### Декоратор для регистрации классов
```python
# Реестр классов
plugins = {}

def register_plugin(name):
    def decorator(cls):
        plugins[name] = cls
        return cls
    return decorator

@register_plugin("json")
class JsonParser:
    def parse(self, data):
        return f"JSON парсинг: {data}"

@register_plugin("csv")
class CsvParser:
    def parse(self, data):
        return f"CSV парсинг: {data}"

# Использование реестра
def parse_file(format, data):
    parser_class = plugins.get(format)
    if not parser_class:
        raise ValueError(f"Неизвестный формат: {format}")
    return parser_class().parse(data)

print(parse_file("json", "{...}"))  # JSON парсинг: {...}
```

### @classmethod
```python
class Date:
    def __init__(self, year, month, day):
        self.year = year
        self.month = month
        self.day = day
    
    @classmethod
    def from_string(cls, date_str):
        year, month, day = map(int, date_str.split('-'))
        return cls(year, month, day)
    
    @classmethod
    def today(cls):
        from datetime import date
        d = date.today()
        return cls(d.year, d.month, d.day)

d1 = Date.from_string("2024-03-12")
d2 = Date.today()
```

### @property
```python
class Circle:
    def __init__(self, radius):
        self._radius = radius
    
    @property
    def radius(self):
        return self._radius
    
    @radius.setter
    def radius(self, value):
        if value < 0:
            raise ValueError("Радиус не может быть отрицательным")
        self._radius = value
    
    @property
    def area(self):
        return 3.14159 * self._radius ** 2
    
    @property
    def diameter(self):
        return self._radius * 2

c = Circle(5)
print(c.radius)  # 5
c.radius = 10    # работает сеттер
print(c.area)    # 314.159
# c.area = 100   # Ошибка! нет сеттера
```

### @cached_property (Python 3.8+)
```python
from functools import cached_property

class DataProcessor:
    def __init__(self, data):
        self.data = data
    
    @cached_property
    def processed(self):
        """Вычисляется один раз и кэшируется"""
        print("Тяжелые вычисления...")
        time.sleep(2)
        return sum(x ** 2 for x in self.data)

dp = DataProcessor([1, 2, 3, 4, 5])
print(dp.processed)  # тяжелые вычисления... 55
print(dp.processed)  # сразу 55 (из кэша)
```

### @dataclass (Python 3.7+)
```python
from dataclasses import dataclass, field

@dataclass
class Person:
    name: str
    age: int
    email: str = field(default="не указан")
    _id: int = field(default=0, repr=False)  # не в __repr__
    
    def __post_init__(self):

            raise ValueError("Возраст не может быть отрицательным")

# Автоматически получаем __init__, __repr__, __eq__

p2 = Person("Анна", 30)
print(p1)          # Person(name='Анна', age=30, email='не указан')
print(p1 == p2)    # True
```

### Декоратор с состоянием
```python
def counter(func):
    """Считает количество вызовов функции"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        wrapper.count += 1
        print(f"Вызов #{wrapper.count} функции {func.__name__}")
        return func(*args, **kwargs)
    
    wrapper.count = 0
    return wrapper

@counter
def say(word):
    print(word)

say("Привет")  # Вызов #1 функции say
say("Пока")    # Вызов #2 функции say
print(say.count)  # 2
```

### Класс-декоратор

```python
class CountCalls:
    def __init__(self, func):
        self.func = func
        self.count = 0
    
    def __call__(self, *args, **kwargs):
        self.count += 1
        print(f"Вызов #{self.count} функции {self.func.__name__}")
```

### Декоратор с параметрами как класс
```python
class Retry:
    def __init__(self, max_attempts=3, delay=1):
        self.max_attempts = max_attempts
        self.delay = delay
    
    def __call__(self, func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(1, self.max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == self.max_attempts:
                        raise
                    print(f"Попытка {attempt} не удалась, пробую снова...")
                    time.sleep(self.delay)
        return wrapper

@Retry(max_attempts=3, delay=0.5)
def unstable_call():
    import random
    if random.random() < 0.7:
        raise ValueError("Ошибка!")
    return "Успех"
```

## Продвинутые концепции

### Метаклассы (кратко)
```python
# Метаклассы создают классы
class SingletonMeta(type):
    """Метакласс для синглтона"""
    _instances = {}
    
    def __call__(cls, *args, **kwargs):
        if cls not in cls._instances:
            cls._instances[cls] = super().__call__(*args, **kwargs)
        return cls._instances[cls]

class Database(metaclass=SingletonMeta):
    def __init__(self):
        print("Создание БД")

db1 = Database()
db2 = Database()
print(db1 is db2)  # True
```

### Дескрипторы
```python
class PositiveNumber:
    """Дескриптор для положительных чисел"""
    def __init__(self, name):
        self.name = name
    
    def __get__(self, obj, objtype=None):
        return obj.__dict__.get(self.name, 0)
    
    def __set__(self, obj, value):
        if value < 0:
            raise ValueError(f"{self.name} должно быть положительным")
        obj.__dict__[self.name] = value

class Order:
    price = PositiveNumber('price')
    quantity = PositiveNumber('quantity')
    
    def __init__(self, price, quantity):
        self.price = price
        self.quantity = quantity
    
    def total(self):
        return self.price * self.quantity

order = Order(100, 2)
print(order.total())  # 200
# order.price = -50  # ValueError!
```

### Контекстные менеджеры через contextlib
```python
from contextlib import contextmanager

@contextmanager
def temporary_change(obj, attr, value):
    """Временно меняет атрибут объекта"""
    old_value = getattr(obj, attr)
    setattr(obj, attr, value)
    try:
        yield
    finally:
        setattr(obj, attr, old_value)

class Config:
    debug = False

config = Config()
print(config.debug)  # False

with temporary_change(config, 'debug', True):
    print(config.debug)  # True

print(config.debug)  # False
```

## Сравнение с C++

| Концепция | C++ | Python |

| Указатели на функции | ✅ | ✅ (функции как объекты) |
