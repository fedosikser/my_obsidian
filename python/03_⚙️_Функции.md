# 03_⚙️_Функции

## Философия функций в Python
- Функции — это **объекты первого класса** (их можно передавать, возвращать, хранить)
- Нет перегрузки по типам аргументов (последнее определение побеждает)
- Поддерживаются именованные аргументы и значения по умолчанию
- Возвращают `None`, если нет `return`

---

## Определение и вызов

### Базовая функция
```python
def greet(name):
    """Документация функции (docstring)"""  # опционально, но желательно
    return f"Привет, {name}!"

# Вызов
result = greet("Анна")
print(result)  # Привет, Анна!
```

### Docstring (документация)
```python
def calculate(a, b, operation="sum"):
    """
    Выполняет математическую операцию над двумя числами.
    
    Аргументы:
        a (int/float): Первое число
        b (int/float): Второе число
        operation (str): Операция ('sum', 'sub', 'mul', 'div')
    
    Возвращает:
        float: Результат операции
    
    Исключения:
        ValueError: Если операция неизвестна
        ZeroDivisionError: При делении на ноль
    """
    if operation == "sum":
        return a + b
    elif operation == "sub":
        return a - b
    # ...

# Посмотреть документацию
help(calculate)
print(calculate.__doc__)
```

## Параметры и аргументы

### Виды параметров (в порядке объявления):
1. **Обычные (позиционные)**
2. `*args` — произвольное число позиционных
3. **Именованные со значением по умолчанию**
4. `**kwargs` — произвольное число именованных

### Позиционные аргументы
```python
def power(base, exponent):
    return base ** exponent

# Порядок важен!
power(2, 3)     # 8
power(3, 2)     # 9 (уже другое!)
```

### Аргументы по умолчанию
```python
def greet(name, greeting="Привет", punctuation="!"):
    return f"{greeting}, {name}{punctuation}"

greet("Анна")                    # "Привет, Анна!"
greet("Анна", "Здравствуй")      # "Здравствуй, Анна!"
greet("Анна", "Hi", "!!!")       # "Hi, Анна!!!"
```

### ВАЖНО! Ловушка с изменяемыми значениями по умолчанию
```python
# НЕПРАВИЛЬНО (значение по умолчанию создается ОДИН РАЗ)
def bad_append(item, lst=[]):    # список создается при определении функции!
    lst.append(item)
    return lst

print(bad_append(1))  # [1]
print(bad_append(2))  # [1, 2]  (оно используется повторно!)
print(bad_append(3))  # [1, 2, 3]

# ПРАВИЛЬНО
def good_append(item, lst=None):
    if lst is None:
        lst = []                 # новый список при каждом вызове
    lst.append(item)
    return lst

print(good_append(1))  # [1]
print(good_append(2))  # [2] (новый список!)
```

### Именованные аргументы (keyword arguments)
```python
def describe_person(name, age, city, job):
    return f"{name}, {age} лет, {city}, {job}"

# Можно указывать в любом порядке, если названы
describe_person(
    city="Москва",
    name="Анна",
    job="Инженер",
    age=30
)

# Смешивать позиционные и именованные
describe_person("Анна", 30, city="Москва", job="Инженер")
# После именованных нельзя ставить позиционные!
# describe_person(name="Анна", 30)  # ОШИБКА!
```

### Произвольное число аргументов (*args)
```python
def sum_all(*args):
    """Суммирует любое количество чисел"""
    print(args)          # кортеж (1, 2, 3, 4)
    return sum(args)

sum_all(1, 2, 3, 4)      # 10
sum_all(1, 2)            # 3
sum_all()                # 0

def print_all(prefix, *items):
    for item in items:
        print(f"{prefix}: {item}")

print_all(">>>", 1, 2, 3, "test")
# >>>: 1
# >>>: 2
# >>>: 3
# >>>: test
```

### Произвольное число именованных аргументов (**kwargs)
```python
def print_info(**kwargs):
    """Печатает все переданные именованные аргументы"""
    for key, value in kwargs.items():
        print(f"{key}: {value}")

print_info(name="Анна", age=30, city="Москва")
# name: Анна
# age: 30
# city: Москва

def configure(**settings):
    """Настройка с значениями по умолчанию"""
    defaults = {
        'host': 'localhost',
        'port': 8080,
        'debug': False
    }
    defaults.update(settings)  # перезаписать значениями из kwargs
    return defaults

configure(port=3000, debug=True)
# {'host': 'localhost', 'port': 3000, 'debug': True}
```

### Все вместе (полный синтаксис)
```python
def complex_function(
    pos1, pos2,           # обычные позиционные
    /,                     # разделитель: ДО него ТОЛЬКО позиционные
    pos_or_kwd1, pos_or_kwd2,  # можно и так, и так
    *,                     # разделитель: ПОСЛЕ него ТОЛЬКО именованные
    kwd1, kwd2="default"   # именованные (со значением по умолчанию)
):
    print(f"pos1={pos1}, pos2={pos2}")
    print(f"pos_or_kwd1={pos_or_kwd1}, pos_or_kwd2={pos_or_kwd2}")
    print(f"kwd1={kwd1}, kwd2={kwd2}")

# Правильные вызовы:
complex_function(1, 2, 3, 4, kwd1=5)
complex_function(1, 2, pos_or_kwd1=3, pos_or_kwd2=4, kwd1=5)

# ОШИБКИ:
# complex_function(pos1=1, pos2=2, 3, 4, kwd1=5)  # pos1 должен быть позиционным
# complex_function(1, 2, 3, 4, 5, kwd2="ok")      # kwd1 пропущен
# complex_function(1, 2, 3, 4, kwd1=5, 6)         # позиционный после именованного
```

## Распаковка аргументов

### Распаковка списка/кортежа в позиционные (*)
```python
def power(base, exponent):
    return base ** exponent

args = (2, 3)
power(*args)          # 8 (распаковка кортежа)

numbers = [1, 2, 3, 4, 5]
sum_all(*numbers)     # sum_all(1, 2, 3, 4, 5)

# Практический пример: split и распаковка
date = "2024-03-12".split("-")
year, month, day = map(int, date)  # 2024, 3, 12
```

### Распаковка словаря в именованные (**)
```python
def describe_person(name, age, city):
    return f"{name}, {age}, {city}"

person = {"name": "Анна", "age": 30, "city": "Москва"}
describe_person(**person)  # "Анна, 30, Москва"

# Комбинирование
defaults = {"host": "localhost", "port": 8080}
user_config = {"port": 3000, "debug": True}
config = {**defaults, **user_config}  # объединение словарей
```

### Практический пример: логирование
```python
def log_message(level, message, **kwargs):
    """Логирует сообщение с дополнительными полями"""
    log_entry = {
        'level': level,
        'message': message,
        'timestamp': datetime.now(),
        **kwargs          # добавить все доп. поля
    }
    print(log_entry)

log_message('INFO', 'Сервер запущен', port=8080, host='localhost')
```

## Возврат значений

### Несколько значений
```python
def get_stats(numbers):
    """Возвращает несколько значений как кортеж"""
    return min(numbers), max(numbers), sum(numbers)/len(numbers)

# Распаковка при получении
minimum, maximum, average = get_stats([1, 2, 3, 4, 5])

# Если нужно только некоторые
_, max_val, _ = get_stats([1, 2, 3, 4, 5])  # игнорируем min и average
```

### Ранний возврат
```python
def divide(a, b):
    if b == 0:
        return None  # или raise ValueError("На ноль делить нельзя")
    return a / b
```

### Возврат None (неявно)
```python
def print_something(x):
    print(x)
    # нет return — возвращает None

result = print_something(5)
print(result)  # None
```

## Области видимости (scope)

### LEGB правило
Поиск переменной идет в порядке:
1. **L**ocal — локальная область функции
2. **E**nclosing — объемлющие функции (замыкания)
3. **G**lobal — глобальная область модуля
4. **B**uilt-in — встроенные имена (print, len и т.д.)

```python
x = 10  # global

def outer():
    y = 20  # enclosing
    
    def inner():
        z = 30  # local
        print(x, y, z)  # видит все!
    
    inner()
```

### global и nonlocal
```python
counter = 0

def increment():
    global counter      # говорим, что используем глобальную
    counter += 1

increment()
print(counter)  # 1

# nonlocal для enclosing scope
def make_counter():
    count = 0
    
    def increment():
        nonlocal count   # берем из enclosing, не local и не global
        count += 1
        return count
    
    return increment

counter1 = make_counter()
print(counter1())  # 1
print(counter1())  # 2
```

## Лямбда-функции (анонимные функции)

### Синтаксис
```python
# lambda аргументы: выражение
square = lambda x: x ** 2
square(5)  # 25

# Несколько аргументов
add = lambda a, b: a + b
add(3, 5)  # 8

# С условием
max_func = lambda a, b: a if a > b else b
```

### Где использовать
```python
# В sort/key
pairs = [(1, 'one'), (3, 'three'), (2, 'two')]
pairs.sort(key=lambda pair: pair[1])  # сортировка по строке

# В map/filter
numbers = [1, 2, 3, 4, 5]
squared = list(map(lambda x: x**2, numbers))
evens = list(filter(lambda x: x % 2 == 0, numbers))

# В min/max с ключом
words = ["python", "java", "c", "javascript"]
longest = max(words, key=lambda w: len(w))  # "javascript"
```

### Когда НЕ использовать лямбды
```python
# ПЛОХО (слишком сложно)
complex_op = lambda x, y: x**2 + 2*x*y + y**2 if x > 0 else y

# ХОРОШО (обычная функция)
def complex_op(x, y):
    if x > 0:
        return x**2 + 2*x*y + y**2
    return y
```

## Функции высшего порядка

### map — применить функцию ко всем элементам
```python
numbers = [1, 2, 3, 4]

# Классический способ
squared = list(map(lambda x: x**2, numbers))  # [1, 4, 9, 16]

# Но лучше list comprehension!
squared = [x**2 for x in numbers]

# С несколькими списками
a = [1, 2, 3]
b = [4, 5, 6]
sums = list(map(lambda x, y: x + y, a, b))  # [5, 7, 9]
```

### filter — отфильтровать элементы
```python
numbers = [1, 2, 3, 4, 5, 6]

evens = list(filter(lambda x: x % 2 == 0, numbers))  # [2, 4, 6]

# List comprehension лучше!
evens = [x for x in numbers if x % 2 == 0]
```

### reduce — свертка списка
```python
from functools import reduce

numbers = [1, 2, 3, 4]
product = reduce(lambda a, b: a * b, numbers)  # 24 (1*2*3*4)
sum_all = reduce(lambda a, b: a + b, numbers)  # 10

# Для суммы есть sum(), для других операций часто есть аналоги
```

### zip — объединение списков
```python
names = ["Анна", "Боб", "Чарли"]
ages = [30, 25, 35]
cities = ["Москва", "СПб", "Казань"]

# Объединение
for name, age, city in zip(names, ages, cities):
    print(f"{name}, {age}, {city}")

# Создание словаря
user_dict = dict(zip(names, ages))  # {'Анна': 30, 'Боб': 25, 'Чарли': 35}

# "Распаковка" zip
pairs = [(1, 'a'), (2, 'b'), (3, 'c')]
numbers, letters = zip(*pairs)  # numbers = (1,2,3), letters = ('a','b','c')
```

## Декораторы (кратко, подробно в [[10_🎨_Декораторы_и_продвинутое]])

### Простейший декоратор
```python
def timer(func):
    """Измеряет время выполнения функции"""
    import time
    
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"{func.__name__} выполнялась {end - start:.4f} сек")
        return result
    
    return wrapper

@timer
def slow_function():
    import time
    time.sleep(1)
    return "Готово"

slow_function()  # slow_function выполнялась 1.0002 сек
```

## Полезные приемы

### Частичное применение (partial)
```python
from functools import partial

def power(base, exponent):
    return base ** exponent

# Зафиксировали exponent=2
square = partial(power, exponent=2)
cube = partial(power, exponent=3)

square(5)  # 25
cube(5)    # 125
```

### Кэширование результатов (мемоизация)
```python
from functools import lru_cache

@lru_cache(maxsize=128)
def fibonacci(n):
    """Фибоначчи с кэшированием"""
    if n < 2:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

# Первый вызов медленный, следующие — мгновенные
fibonacci(100)
```

### Проверка аргументов
```python
def set_age(age):
    if not isinstance(age, int):
        raise TypeError(f"Возраст должен быть int, а не {type(age).__name__}")
    if age < 0 or age > 150:
        raise ValueError(f"Возраст {age} вне допустимого диапазона")
    print(f"Возраст установлен: {age}")
```

## Сравнение с C++

| C++ | Python |
|-----|--------|
| `int add(int a, int b)` | `def add(a, b):` |
| Перегрузка функций | Не поддерживается |
| `int&` для изменения | Возврат кортежа |
| Шаблоны | Динамическая типизация |
| Лямбды | `lambda` |
| `std::bind` | `functools.partial` |
